```python
import tkinter as tk
import random

# Naive AI class to randomly select Rock, Paper, or Scissors
class NaiveAI:
    def __init__(self):
        self.choices = ["Rock", "Paper", "Scissors"]
    
    def make_choice(self):
        return random.choice(self.choices)

# Game Logic class to evaluate the outcomes
class GameLogic:
    def __init__(self):
        self.ai = NaiveAI()
    
    def evaluate_game(self, player_choice):
        ai_choice = self.ai.make_choice()
        result = self.determine_winner(player_choice, ai_choice)
        return ai_choice, result
    
    def determine_winner(self, player_choice, ai_choice):
        if player_choice == ai_choice:
            return "Draw"
        elif (player_choice == "Rock" and ai_choice == "Scissors") or \
             (player_choice == "Scissors" and ai_choice == "Paper") or \
             (player_choice == "Paper" and ai_choice == "Rock"):
            return "Win"
        else:
            return "Lose"

# Main GUI class for the Rock-Paper-Scissors Game
class RockPaperScissorsGame:
    def __init__(self, master):
        self.master = master
        self.master.title("Rock-Paper-Scissors Game")
        
        self.game_logic = GameLogic()

        # Title Label
        self.title_label = tk.Label(master, text="Rock-Paper-Scissors", font=("Arial", 24))
        self.title_label.pack(pady=10)

        # Create choice buttons
        self.create_choice_buttons()

        # Display Areas
        self.player_choice_label = tk.Label(master, text="Your Choice: ")
        self.player_choice_label.pack(pady=5)

        self.ai_choice_label = tk.Label(master, text="AI Choice: ")
        self.ai_choice_label.pack(pady=5)

        self.result_label = tk.Label(master, text="", font=("Arial", 16))
        self.result_label.pack(pady=20)

        # Replay Button
        replay_button = tk.Button(master, text="Replay", command=self.reset_game)
        replay_button.pack(pady=10)

    def create_choice_buttons(self):
        button_frame = tk.Frame(self.master)
        button_frame.pack(pady=10)

        rock_button = tk.Button(button_frame, text="Rock", command=lambda: self.player_choice("Rock"))
        rock_button.pack(side=tk.LEFT, padx=10)

        paper_button = tk.Button(button_frame, text="Paper", command=lambda: self.player_choice("Paper"))
        paper_button.pack(side=tk.LEFT, padx=10)

        scissors_button = tk.Button(button_frame, text="Scissors", command=lambda: self.player_choice("Scissors"))
        scissors_button.pack(side=tk.LEFT, padx=10)

    def player_choice(self, choice):
        ai_choice, result = self.game_logic.evaluate_game(choice)
        self.update_gui(choice, ai_choice, result)

    def update_gui(self, player_choice, ai_choice, outcome):
        self.player_choice_label.config(text=f"Your Choice: {player_choice}")
        self.ai_choice_label.config(text=f"AI Choice: {ai_choice}")
        self.result_label.config(text=outcome)

    def reset_game(self):
        self.player_choice_label.config(text="Your Choice: ")
        self.ai_choice_label.config(text="AI Choice: ")
        self.result_label.config(text="")

if __name__ == "__main__":
    root = tk.Tk()
    game = RockPaperScissorsGame(root)
    root.mainloop()
```

### Instructions for Running the Game
1. Ensure you have Python 3.x installed on your system.
2. Copy the above code into a file named `rock_paper_scissors.py`.
3. Run the script using the command:
   ```bash
   python rock_paper_scissors.py
   ```
4. A window will open displaying the game. Click on "Rock", "Paper", or "Scissors" to play against the AI. The result will be displayed after each round, and you can press the "Replay" button to start a new round.
