```python
import tkinter as tk
import random

class RockPaperScissorsGame:
    def __init__(self, root):
        self.root = root
        self.root.title("Rock-Paper-Scissors Game")
        self.root.geometry("400x300")

        self.player_choice = ""
        self.ai_choice = ""
        
        # Choice Buttons
        self.rock_button = tk.Button(root, text="Rock", command=lambda: self.play("Rock"), width=10, height=2)
        self.paper_button = tk.Button(root, text="Paper", command=lambda: self.play("Paper"), width=10, height=2)
        self.scissors_button = tk.Button(root, text="Scissors", command=lambda: self.play("Scissors"), width=10, height=2)

        self.rock_button.pack(pady=10)
        self.paper_button.pack(pady=10)
        self.scissors_button.pack(pady=10)

        # Display Areas
        self.player_label = tk.Label(root, text="Your Choice: ")
        self.player_label.pack(pady=5)
        
        self.ai_label = tk.Label(root, text="AI Choice: ")
        self.ai_label.pack(pady=5)

        self.result_label = tk.Label(root, text="Result: ")
        self.result_label.pack(pady=5)

        # Reset Button
        self.reset_button = tk.Button(root, text="Reset", command=self.reset_game, width=10, height=2)
        self.reset_button.pack(pady=20)

    def play(self, player_choice):
        self.player_choice = player_choice
        self.ai_choice = random.choice(["Rock", "Paper", "Scissors"])
        self.update_display()
        
    def update_display(self):
        self.player_label.config(text=f"Your Choice: {self.player_choice}")
        self.ai_label.config(text=f"AI Choice: {self.ai_choice}")
        result = self.determine_winner()
        self.result_label.config(text=f"Result: {result}")

    def determine_winner(self):
        if self.player_choice == self.ai_choice:
            return "Draw"
        elif (self.player_choice == "Rock" and self.ai_choice == "Scissors") or \
             (self.player_choice == "Paper" and self.ai_choice == "Rock") or \
             (self.player_choice == "Scissors" and self.ai_choice == "Paper"):
            return "You Win!"
        else:
            return "You Lose!"

    def reset_game(self):
        self.player_choice = ""
        self.ai_choice = ""
        self.player_label.config(text="Your Choice: ")
        self.ai_label.config(text="AI Choice: ")
        self.result_label.config(text="Result: ")

if __name__ == "__main__":
    root = tk.Tk()
    game = RockPaperScissorsGame(root)
    root.mainloop()
```

### Explanation of the Code
1. **Imports**: The script imports the necessary libraries, `tkinter` for GUI and `random` for AI selection.
2. **Class Definition**: The `RockPaperScissorsGame` class encapsulates the game's logic and UI.
3. **Initialization**: The `__init__` method sets up the main window, buttons for player choices, display labels, and a reset button.
4. **Play Method**: The `play` method takes the player's choice, randomly selects the AI's choice, and updates the display.
5. **Update Display**: The `update_display` method updates the labels with both the player's choice, AI's choice, and the result.
6. **Determine Winner**: The `determine_winner` method evaluates the choices according to game rules and returns the result.
7. **Reset Game**: The `reset_game` method resets all labels and choices for a new game.
8. **Main Execution**: The script checks if it's run as the main program and initiates the game interface.

This code provides a fully functional Rock-Paper-Scissors game with a user-friendly interface, enabling players to compete against a naive AI.
