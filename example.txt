```python
import tkinter as tk
import random

class RockPaperScissorsGame:
    def __init__(self, master):
        self.master = master
        self.master.title("Rock-Paper-Scissors Game")
        self.master.geometry("400x300")
        self.master.configure(bg="#f0f0f0")

        self.player_choice = None
        self.ai_choice = None
        self.result = None

        # Title label
        self.title_label = tk.Label(master, text="Rock-Paper-Scissors", font=("Arial", 20, "bold"), bg="#f0f0f0")
        self.title_label.pack(pady=20)

        # Player choice section
        self.player_choice_label = tk.Label(master, text="Your Choice:", bg="#f0f0f0")
        self.player_choice_label.pack()
        self.player_choice_display = tk.Label(master, text="None", bg="#f0f0f0")
        self.player_choice_display.pack()

        # AI choice section
        self.ai_choice_label = tk.Label(master, text="AI Choice:", bg="#f0f0f0")
        self.ai_choice_label.pack()
        self.ai_choice_display = tk.Label(master, text="None", bg="#f0f0f0")
        self.ai_choice_display.pack()

        # Result section
        self.result_label = tk.Label(master, text="Result:", bg="#f0f0f0")
        self.result_label.pack()
        self.result_display = tk.Label(master, text="Not determined yet", bg="#f0f0f0")
        self.result_display.pack()

        # Buttons
        self.rock_button = tk.Button(master, text="Rock", command=lambda: self.select_choice('Rock'))
        self.rock_button.pack(side=tk.LEFT, padx=20, pady=20)
        self.paper_button = tk.Button(master, text="Paper", command=lambda: self.select_choice('Paper'))
        self.paper_button.pack(side=tk.LEFT, padx=20, pady=20)
        self.scissors_button = tk.Button(master, text="Scissors", command=lambda: self.select_choice('Scissors'))
        self.scissors_button.pack(side=tk.LEFT, padx=20, pady=20)

    def select_choice(self, choice):
        self.player_choice = choice
        self.player_choice_display.config(text=self.player_choice)
        self.ai_choice = self.ai_random_selection()
        self.ai_choice_display.config(text=self.ai_choice)
        self.result = self.compare_choices()
        self.result_display.config(text=self.result)

    def ai_random_selection(self):
        return random.choice(['Rock', 'Paper', 'Scissors'])

    def compare_choices(self):
        if self.player_choice == self.ai_choice:
            return "draw"
        elif (self.player_choice == 'Rock' and self.ai_choice == 'Scissors') or \
             (self.player_choice == 'Paper' and self.ai_choice == 'Rock') or \
             (self.player_choice == 'Scissors' and self.ai_choice == 'Paper'):
            return "win"
        else:
            return "lose"

if __name__ == "__main__":
    root = tk.Tk()
    game = RockPaperScissorsGame(root)
    root.mainloop()
```
