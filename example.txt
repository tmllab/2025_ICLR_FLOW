\section{Algorithm 1: Q-Learning}
\begin{frame}
  \frametitle{Q-Learning}
  \textbf{Motivation:} 
  Q-Learning is a model-free reinforcement learning algorithm that learns the value of an action in a particular state.
  
  \textbf{Problem:} 
  How can an agent learn to act optimally in an unknown environment?
  
  \textbf{Intuitive Solution:} 
  The agent learns from the consequences of its actions using a Q-value update rule.
  
  \textbf{Mathematical Equation:}
  \[
  Q(s, a) \leftarrow Q(s, a) + \alpha \left( r + \gamma \max_{a'} Q(s', a') - Q(s, a) \right)
  \]
\end{frame}

\section{Algorithm 2: SARSA}
\begin{frame}
  \frametitle{SARSA}
  \textbf{Motivation:} 
  SARSA is an on-policy algorithm where the agent learns the value of the policy being followed.
  
  \textbf{Problem:} 
  How does the agent improve its policy while learning?
  
  \textbf{Intuitive Solution:} 
  The agent updates its Q-values based on the actions it actually takes.
  
  \textbf{Mathematical Equation:}
  \[
  Q(s, a) \leftarrow Q(s, a) + \alpha \left( r + \gamma Q(s', a') - Q(s, a) \right)
  \]
\end{frame}

\section{Algorithm 3: Deep Q-Network (DQN)}
\begin{frame}
  \frametitle{Deep Q-Network (DQN)}
  \textbf{Motivation:} 
  DQN extends Q-learning using deep neural networks to approximate Q-values.
  
  \textbf{Problem:} 
  How can we handle high-dimensional state spaces?
  
  \textbf{Intuitive Solution:} 
  By using a neural network to generalize Q-values across similar states.
  
  \textbf{Mathematical Equation:}
  \[
  \text{Loss} = (y - Q(s, a; \theta))^2 \quad \text{where} \quad y = r + \gamma \max_{a'} Q(s', a'; \theta^-)
  \]
\end{frame}
