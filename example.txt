import tkinter as tk
import random

def evaluate_outcome(player_choice, ai_choice):
    """
    This function evaluates the outcome of the game based on the player's choice 
    and the AI's choice.
    
    Parameters:
        player_choice (str): The choice made by the player ('Rock', 'Paper', 'Scissors').
        ai_choice (str): The choice made by the AI ('Rock', 'Paper', 'Scissors').
        
    Returns:
        str: The result of the game, which can be 'You Win!', 'You Lose!', or 'It's a Draw!'.
    """
    if player_choice == ai_choice:
        return "It's a Draw!"
    elif (player_choice == 'Rock' and ai_choice == 'Scissors') or \
         (player_choice == 'Paper' and ai_choice == 'Rock') or \
         (player_choice == 'Scissors' and ai_choice == 'Paper'):
        return "You Win!"
    else:
        return "You Lose!"

def naive_ai_choice():
    """
    This function simulates a naive AI that randomly selects 
    Rock, Paper, or Scissors.
    
    Returns:
        str: The AI's choice, which can be 'Rock', 'Paper', or 'Scissors'.
    """
    return random.choice(['Rock', 'Paper', 'Scissors'])

def process_game(player_choice):
    """
    This function processes the game by capturing the player's choice, invoking 
    the AI to select its choice, evaluating the outcome, and returning the result 
    along with the AI's choice.
    
    Parameters:
        player_choice (str): The choice made by the player ('Rock', 'Paper', 'Scissors').
    
    Returns:
        tuple: A tuple containing the AI's choice and the result of the game.
    """
    ai_choice = naive_ai_choice()  # Invoke the naive AI to get its choice
    result = evaluate_outcome(player_choice, ai_choice)  # Evaluate the outcome
    return ai_choice, result  # Return both the AI's choice and the evaluation result

def player_choice(choice):
    """
    This function is called when the player makes a choice. It processes the choice,
    gets the AI's choice, and updates the UI with the results.
    """
    ai_choice, result = process_game(choice)  # Call the game logic to process the player's choice
    ai_choice_label.config(text=f"AI Choice: {ai_choice}")  # Update the UI with the AI's choice
    result_label.config(text=result)  # Update the UI with the game result

def play_again():
    """
    This function resets the game state for a new round when the player clicks 'Play Again'.
    """
    ai_choice_label.config(text="AI Choice: ")  # Clear the AI choice display
    result_label.config(text="Result: ")  # Clear the result display

# Setting up the main window
root = tk.Tk()
root.title("Rock-Paper-Scissors Game")
root.geometry("300x200")

# Title label
title_label = tk.Label(root, text="Rock-Paper-Scissors Game", font=("Helvetica", 16))
title_label.pack(pady=10)

# Player choice buttons
rock_button = tk.Button(root, text="Rock", command=lambda: player_choice('Rock'))
rock_button.pack(side=tk.LEFT, padx=10)

paper_button = tk.Button(root, text="Paper", command=lambda: player_choice('Paper'))
paper_button.pack(side=tk.LEFT, padx=10)

scissors_button = tk.Button(root, text="Scissors", command=lambda: player_choice('Scissors'))
scissors_button.pack(side=tk.LEFT, padx=10)

# AI choice display
ai_choice_label = tk.Label(root, text="AI Choice: ")
ai_choice_label.pack(pady=10)

# Result display
result_label = tk.Label(root, text="Result: ")
result_label.pack(pady=10)

# Play Again button
play_again_button = tk.Button(root, text="Play Again", command=play_again)
play_again_button.pack(pady=10)

# Run the application
root.mainloop()

#### 1. Overview of the Game

The Rock-Paper-Scissors game is a classic hand game played with three possible moves: rock, paper, and scissors. The player competes against a naive AI that randomly selects one of the three options. The game displays the player’s choice, the AI’s choice, and the result of each round (win, lose, or draw).

#### 2. Prerequisites

Before running the game, ensure that the following dependencies are installed:

- **Python**: Version 3.6 or higher is required. Python can be downloaded from the official website: [python.org](https://www.python.org/downloads/).

- **Tkinter**: This library is included with standard Python installations on Windows and macOS. For Linux, ensure that it is installed using the package manager (e.g., `sudo apt-get install python3-tk` for Debian-based systems).

To run the Rock-Paper-Scissors game, follow these steps:
